fn get(&self, text: &[u8]) -> Vec<u8>

Calculate the reverse complement of given text. The text has to be in DNA alphabet containing only ACGTacgt symbols. Other symbols won't be converted.

Example

use bio::alphabets::dna::RevComp;
let revcomp = RevComp::new();
let text = b"AAACCTT";
let revcomp_text = revcomp.get(text);
assert_eq!(revcomp_text, &b"AAGGTTT"[..]);
assert_eq!(revcomp.get(&revcomp_text[..]), &text[..]);
